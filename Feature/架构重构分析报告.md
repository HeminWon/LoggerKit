#### 正确的实现方案

**方案 1：使用 Actor（推荐）**

actor LoggerEngine {
    static let shared = LoggerEngine()

    private var destinations: [BaseDestination] = []
    private var databaseManager: LogDatabaseManager?

    func addDestination(_ destination: BaseDestination) {
        destinations.append(destination)  // ✅ Actor 自动串行化
    }
}

**方案 2：使用锁（次选）**

```swift
class LoggerEngine {
    static let shared = LoggerEngine()

    private let lock = NSLock()
    private var destinations: [BaseDestination] = []

    func addDestination(_ destination: BaseDestination) {
        lock.lock()
        defer { lock.unlock() }
        destinations.append(destination)  // ✅ 锁保护
    }
}
```

**方案 3：使用不可变状态**

```swift
class LoggerEngine {
    static let shared = LoggerEngine()

    private let destinations: [BaseDestination]  // ✅ let 不可变

    init(destinations: [BaseDestination] = []) {
        self.destinations = destinations
    }
}

extension LoggerEngine: Sendable {}  // ✅ 真正的 Sendable
```

#### 统一缓存方案

**泛型 ConcurrentCache**

```swift
protocol CacheProtocol {
    associatedtype Key: Hashable
    associatedtype Value

    func get(_ key: Key) -> Value?
    func set(_ key: Key, value: Value)
    func remove(_ key: Key)
    func clear()
}

final class ConcurrentCache<K: Hashable, V>: CacheProtocol {
    private var cache: [K: V] = [:]
    private let queue = DispatchQueue(label: "cache", attributes: .concurrent)

    func get(_ key: K) -> V? {
        queue.sync { cache[key] }
    }

    func set(_ key: K, value: V) {
        queue.async(flags: .barrier) {
            self.cache[key] = value
        }
    }

    func remove(_ key: K) {
        queue.async(flags: .barrier) {
            self.cache.removeValue(forKey: key)
        }
    }

    func clear() {
        queue.async(flags: .barrier) {
            self.cache.removeAll()
        }
    }
}
```

**使用示例**

```swift
// ✅ FilterOptionsCache 基于 ConcurrentCache
class FilterOptionsCache {
    private let cache = ConcurrentCache<String, [String]>()

    func getOptions(for key: String) -> [String]? {
        cache.get(key)
    }

    func setOptions(_ options: [String], for key: String) {
        cache.set(key, value: options)
    }
}

// ✅ SearchState 也使用统一缓存
class SearchState {
    private let cache = ConcurrentCache<String, SearchResults>()
}
```

@Observable
final class SearchViewModel {
    var searchText: String = ""
    private(set) var cachedResults: SearchResults?
    private(set) var isSearching = false

    var onSearchChanged: (() -> Void)?

    private var searchTask: Task<Void, Never>?
    private var cancellables = Set<AnyCancellable>()

    init() {
        setupObservers()
    }

    private func setupObservers() {
        $searchText
            .debounce(for: .milliseconds(100), scheduler: DispatchQueue.main)
            .sink { [weak self] _ in
                self?.onSearchChanged?()
            }
            .store(in: &cancellables)
    }

    func computeResults(from events: [LogEvent]) async {
        searchTask?.cancel()

        guard !searchText.isEmpty else {
            await MainActor.run {
                cachedResults = nil
            }
            return
        }

        isSearching = true
        defer { isSearching = false }

        searchTask = Task.detached { [weak self] in
            guard let self else { return }
            let results = self.performSearch(in: events)

            await MainActor.run {
                self.cachedResults = results
            }
        }

        await searchTask?.value
    }

    private func performSearch(in events: [LogEvent]) -> SearchResults {
        // 单次遍历计算所有匹配
        var results = SearchResults()
        let keyword = searchText.lowercased()

        for event in events {
            if event.message.lowercased().contains(keyword) {
                results.messages.append(event)
            }
            if event.fileName.lowercased().contains(keyword) {
                results.fileNames.append(event)
            }
            // ...
        }

        return results
    }

    func clear() {
        searchText = ""
        cachedResults = nil
        searchTask?.cancel()
    }
}

// ExportUseCase.swift (~100行)
final class ExportUseCase {
    private let databaseManager: LogDatabaseManagerProtocol
    private let parser: LogParser

    init(
        databaseManager: LogDatabaseManagerProtocol,
        parser: LogParser = LogParser()
    ) {
        self.databaseManager = databaseManager
        self.parser = parser
    }

    func exportLogs(
        withFilter filter: FilterViewModel,
        format: ExportFormat = .txt
    ) async throws -> URL {
        // 流式导出逻辑
        let fileURL = try createTempFile(format: format)

        var offset = 0
        let batchSize = 1000

        while true {
            let events = try await databaseManager.fetchEvents(
                offset: offset,
                limit: batchSize,
                // 应用过滤条件
                levels: filter.selectedLevels,
                functions: filter.selectedFunctions
                // ...
            )

            if events.isEmpty { break }

            let content = formatEvents(events, format: format)
            try appendToFile(content, at: fileURL)

            offset += batchSize
        }

        return fileURL
    }

    private func createTempFile(format: ExportFormat) throws -> URL {
        let fileName = "logs_\(Date().timeIntervalSince1970).\(format.fileExtension)"
        return FileManager.default.temporaryDirectory
            .appendingPathComponent(fileName)
    }

    private func formatEvents(_ events: [LogEvent], format: ExportFormat) -> String {
        switch format {
        case .txt:
            return parser.formatAsText(events)
        case .json:
            return parser.formatAsJSON(events)
        case .csv:
            return parser.formatAsCSV(events)
        }
    }
}

enum ExportFormat {
    case txt, json, csv

    var fileExtension: String {
        switch self {
        case .txt: return "txt"
        case .json: return "json"
        case .csv: return "csv"
        }
    }
}

// LoggerEngine.swift
actor LoggerEngine {
    static let shared = LoggerEngine()

    private var destinations: [BaseDestination] = []
    private var databaseManager: LogDatabaseManager?
    private var rotationManager: LogDatabaseRotationManager?

    private init() {
        // 延迟初始化
    }

    // ✅ Actor 方法自动串行化
    func configure(
        enableConsole: Bool = true,
        enableDatabase: Bool = true
    ) {
        if enableConsole {
            let console = ConsoleDestination()
            destinations.append(console)
        }

        if enableDatabase {
            let coreDataDestination = CoreDataDestination()
            destinations.append(coreDataDestination)

            databaseManager = LogDatabaseManager()
            rotationManager = LogDatabaseRotationManager(databaseManager: databaseManager!)
        }
    }

    func log(
        _ message: String,
        level: SwiftyBeaver.Level,
        file: String,
        function: String,
        line: Int
    ) {
        // ✅ 线程安全
        for destination in destinations {
            destination.send(level, msg: message, file: file, function: function, line: line)
        }
    }

    func getDatabaseManager() -> LogDatabaseManager? {
        databaseManager
    }
}

// ✅ 无需 @unchecked Sendable，Actor 自动符合
