#### 1.1 LoggerEngine 的并发安全问题

**文件**: `Sources/LoggerKit/Core/LoggerEngine.swift`

**问题描述**:
- `@unchecked Sendable` 掩盖数据竞争
- `ensureConfigured()` 存在 TOCTOU 漏洘（L100-106）
- `isConfigured` 先读后锁，可能导致重复初始化

**当前代码**:
```swift
private func ensureConfigured() {
    lock.lock()                    // ❌ 锁太晚
    let configured = isConfigured  // ⚠️ 先读取未锁保护的变量
    lock.unlock()

    if !configured {               // ❌ 可能多次调用 configure
        configure(LoggerEngineConfiguration())
    }
}
```

**修复方案**:
```swift
// 使用 DispatchQueue 同步确保原子性
private let lockQueue = DispatchQueue(label: "com.loggerkit.engine", qos: .userInitiated)
private var _isConfigured = false

private func ensureConfigured() {
    lockQueue.sync {
        guard !_isConfigured else { return }
        configure(LoggerEngineConfiguration())
    }
}

// configure() 方法也需要在 lockQueue 中执行
public func configure(_ configuration: LoggerEngineConfiguration) {
    lockQueue.sync {
        guard !_isConfigured else {
            assertionFailure("LoggerEngine 已配置，不能重复配置")
            return
        }
        setupDestinations(configuration)
        _isConfigured = true
    }
}
```

**影响范围**: 整个日志系统的初始化

#### 1.2 CoreDataDestination 的数据竞争

**文件**: `Sources/LoggerKit/Database/CoreDataDestination.swift`

**问题描述**:
- `pendingEvents` 在 `queue` 上访问，但无显式同步
- `addEvent()` 和 `flushPendingEvents()` 并发访问同一数组
- `debounceTimer` 的 handler 可能在不同线程调用

**当前代码**:
```swift
private func addEvent(_ event: LogEvent) {
    pendingEvents.append(event)  // ❌ 可能并发访问

    if immediateFlushLevels.contains(event.level) {
        flushPendingEvents()     // ⚠️ 访问 pendingEvents
        return
    }

    if pendingEvents.count >= batchSize {  // ❌ 再次访问
        flushPendingEvents()
        return
    }

    scheduleDebounceFlush()
}
```

**修复方案**:
```swift
// 确保所有访问都在 queue 上（已经在 L83 异步调用）
// 但需要确保 flushPendingEvents 也总是在 queue 上调用

// 修改 flush() 方法，保持一致性
public func flush() {
    queue.async { [weak self] in
        self?.flushPendingEvents()
    }
}

// scheduleDebounceFlush 已经正确使用 queue
// 需要添加断言确保线程正确
private func addEvent(_ event: LogEvent) {
    dispatchPrecondition(condition: .onQueue(queue))  // ✅ 添加线程检查

    pendingEvents.append(event)

    if immediateFlushLevels.contains(event.level) {
        flushPendingEvents()
        return
    }

    if pendingEvents.count >= batchSize {
        flushPendingEvents()
        return
    }

    scheduleDebounceFlush()
}

private func flushPendingEvents() {
    dispatchPrecondition(condition: .onQueue(queue))  // ✅ 添加线程检查

    debounceTimer?.cancel()
    debounceTimer = nil

    guard !pendingEvents.isEmpty else { return }

    let eventsToWrite = pendingEvents
    pendingEvents.removeAll(keepingCapacity: true)

    // 后台上下文批量写入
    let context = coreDataStack.newBackgroundContext()

    context.perform {
        for event in eventsToWrite {
            _ = LogEventEntity.create(from: event, in: context)
        }

        do {
            try context.save()
        } catch {
            print("❌ CoreDataDestination: Failed to save logs: \(error)")
        }
    }
}
```

**影响范围**: 所有日志写入操作

#### 1.3 CoreDataDestination.deinit 死锁风险

**文件**: `Sources/LoggerKit/Database/CoreDataDestination.swift`

**问题描述**:
- `deinit` 中使用 `queue.sync` 可能死锁
- 如果 `deinit` 在 `queue` 线程上被调用，会永久阻塞

**当前代码**:
```swift
deinit {
    debounceTimer?.cancel()
    debounceTimer = nil

    queue.sync {  // ❌ 可能死锁
        flushPendingEvents()
    }
}
```

**修复方案**:
```swift
deinit {
    debounceTimer?.cancel()
    debounceTimer = nil

    // 使用异步 + 信号量，设置超时
    let semaphore = DispatchSemaphore(value: 0)

    queue.async { [weak self] in
        self?.flushPendingEvents()
        semaphore.signal()
    }

    // 等待最多 1 秒，避免阻塞主线程
    let timeout = DispatchTime.now() + 1.0
    if semaphore.wait(timeout: timeout) == .timedOut {
        print("⚠️ CoreDataDestination: deinit flush timeout, some logs may be lost")
    }
}
```

**影响范围**: App 退出或组件销毁时

#### 1.4 LogDatabaseManager 批量删除的异常处理

**文件**: `Sources/LoggerKit/Database/LogDatabaseManager.swift`

**问题描述**:
- `performAndWait` 内部的 `do-catch` 吞掉异常
- 外层的 `throws` 签名无效
- 强制类型转换 `as!` 可能崩溃

**当前代码**:
```swift
public func deleteLogs(forDate date: String) throws {
    let context = coreDataStack.newBackgroundContext()

    context.performAndWait {
        // ...
        do {
            let result = try context.execute(deleteRequest) as! NSBatchDeleteResult
            let objectIDs = result.result as! [NSManagedObjectID]
            // ...
        } catch {
            print("❌ Failed to delete logs: \(error)")
            // ⚠️ 错误被吞掉，外层 throws 不会抛出
        }
    }
}
```

**修复方案**:
```swift
// 定义自定义错误类型
public enum LogDatabaseError: Error {
    case invalidDeleteResult
    case deleteFailed(Error)
}

// 修复删除方法（应用到所有三个删除方法）
public func deleteLogs(forDate date: String) throws {
    let context = coreDataStack.newBackgroundContext()
    var thrownError: Error?

    context.performAndWait {
        let fetchRequest = LogEventEntity.fetchRequest()
        fetchRequest.predicate = NSPredicate(format: "date == %@", date)

        let deleteRequest = NSBatchDeleteRequest(fetchRequest: fetchRequest as! NSFetchRequest<NSFetchRequestResult>)
        deleteRequest.resultType = .resultTypeObjectIDs

        do {
            guard let result = try context.execute(deleteRequest) as? NSBatchDeleteResult,
                  let objectIDs = result.result as? [NSManagedObjectID] else {
                throw LogDatabaseError.invalidDeleteResult
            }

            NSManagedObjectContext.mergeChanges(
                fromRemoteContextSave: [NSDeletedObjectsKey: objectIDs],
                into: [coreDataStack.viewContext]
            )

            print("✅ Deleted \(objectIDs.count) logs for date \(date)")
        } catch {
            thrownError = error
        }
    }

    if let error = thrownError {
        throw error
    }
}

// 同样修复：
// - deleteLogs(before:) (L457-485)
// - deleteLogs(forSession:) (L515-546)
// - deleteLogs(forSessions:) (L549-582)
```

**影响范围**: 数据库清理、轮转、用户手动删除

#### 1.5 FilterOptionsCache 的 weak self 导致数据丢失

**文件**: `Sources/LoggerKit/UI/FilterOptionsCache.swift`

**问题描述**:
- `queue.sync(flags: .barrier)` 中使用 `[weak self]`
- 如果 self 被释放，写入操作静默失败

**当前代码**:
```swift
func invalidateAll() {
    queue.sync(flags: .barrier) { [weak self] in  // ⚠️ weak self
        self?.storage.removeAll()  // ❌ 可能静默失败
    }
}
```

**修复方案**:
```swift
// 移除 weak self，使用强引用（缓存应保证操作原子性）
func invalidateAll() {
    queue.sync(flags: .barrier) {
        self.storage.removeAll()
    }
}

private func write(_ key: CacheKey, value: Any) {
    queue.sync(flags: .barrier) {
        self.storage[key] = value
    }
}

func invalidate(_ key: CacheKey) {
    queue.sync(flags: .barrier) {
        self.storage[key] = nil
    }
}
```

**影响范围**: 过滤选项的缓存一致性

#### 1.6 CoreDataStack.modelURL 的 fatalError

**文件**: `Sources/LoggerKit/Database/CoreDataStack.swift`

**问题描述**:
- 找不到 CoreData 模型时直接崩溃
- 没有降级策略

**当前代码**:
```swift
private static let modelURL: URL = {
    // ... 查找逻辑 ...
    fatalError("Failed to find LoggerKit CoreData model in bundle...")
}()
```

**修复方案**:
```swift
// 方案 1: 返回 Optional，延迟初始化
private static var _modelURL: URL?
private static func findModelURL() -> URL? {
    let candidatePaths: [(resource: String, extension: String)] = [
        ("LoggerKit", "momd"),
        ("LoggerKit", "mom"),
        ("LoggerKit", "xcdatamodeld")
    ]

    for (resource, ext) in candidatePaths {
        if let url = Bundle.module.url(forResource: resource, withExtension: ext) {
            return url
        }
    }

    print("❌ CoreData model not found, database feature will be disabled")
    return nil
}

// 在 init 中处理
public init() {
    guard let modelURL = Self.findModelURL(),
          let model = NSManagedObjectModel(contentsOf: modelURL) else {
        // 降级：禁用数据库功能
        fatalError("LoggerKit database initialization failed. Please check bundle resources.")
    }

    // ... 继续初始化 ...
}

// 方案 2: 在 LoggerEngine.configure 中提供回退
// 如果数据库不可用，只启用 console logging
```
